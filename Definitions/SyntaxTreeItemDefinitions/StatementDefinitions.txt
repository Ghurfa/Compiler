>Statement
+Statement
{
    .CodeBlock
    OpenCurly
    Statement[]:Statements;untilPop CloseCurly

    $Statement
    {
        .Empty
        Semicolon

        .Expression
        Expression
        ;throw InvalidStatement Expression;ifNotType Expression ICompleteStatement
        Semicolon?;EnsureValidStatementEnding

        .If
        IfKeyword
        Expression:Condition
        Statement:IfTrue
        ElseKeyword?
        Statement:IfFalse;ifNotNull ElseKeyword
        ;EnsureWhitespaceAfter IfKeyword;ifNotType Condition PerenthesizedExpression
        ;EnsureLineBreakAfter Condition;ifNotType Condition PerenthesizedExpression;ifNotType IfTrue CodeBlock
        ;EnsureWhitespaceAfter ElseKeyword;ifNotNull ElseKeyword;ifNotType IfFalse CodeBlock

        .While
        WhileKeyword
        Expression:Condition
        Statement:Body
        ;EnsureWhitespaceAfter WhileKeyword;ifNotType Condition PerenthesizedExpression
        ;EnsureLineBreakAfter Condition;ifNotType Condition PerenthesizedExpression;ifNotType Body CodeBlock

        .For
        ForKeyword
        OpenPeren
        Statement:StartStatement
        Expression:ContinueExpr
        Semicolon?:SecondSemicolon;EnsureValidStatementEnding
        Statement:RepeatStatement
        ClosePeren
        Statement:Body

        .Foreach
        ForeachKeyword
        OpenPeren
        Expression:IteratorDecl
        InKeyword
        Expression:IterateOn
        ClosePeren
        Statement:Body

        .Switch
        SwitchKeyword
        Expression:SwitchOn
        OpenBracket
        CaseStatement[];untilPop CloseBracket
        ;EnsureWhitespaceAfter SwitchKeyword;IfNotType SwitchOn PerenthesizedExpression

        .Return
        ReturnKeyword
        Expression
        ;EnsureWhitespaceAfter ReturnKeyword

        .Exit
        ExitKeyword
        Semicolon?;EnsureValidStatementEnding

        .Break
        BreakKeyword
        Semicolon?;EnsureValidStatementEnding

        .Continue
        ContinueKeyword
        Semicolon?;EnsureValidStatementEnding

        .Throw
        ThrowKeyword
        Expression:Exception
        Semicolon?;EnsureValidStatementEnding
    }
}
