{
    FizzBuzz class
    {
        FizzBuzz public static void(start:int, end:int)
        {
            for(i := start; i < end; i++)
            {
                fizz := i % 3 == 0
                buzz := i % 5 == 0
                fizzStr := "fizz"
                buzzStr := "buzz"
                print:string = fizz && buzz ? "FizzBuzz" \ fizz ? fizzStr \ buzz ? buzzStr \ i.ToString()
            }
        }
    }
